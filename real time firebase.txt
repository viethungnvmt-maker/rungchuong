// game-manager.js
import { database, ref, set, onValue, push, update, remove } from './firebase-config.js';

class GameManager {
  constructor() {
    this.currentRoomId = null;
    this.currentUserId = null;
    this.playerName = '';
    this.playerTeam = '';
  }

  // TẠO PHÒNG
  async createRoom(playerName, team, settings) {
    const roomsRef = ref(database, 'rooms');
    const newRoomRef = push(roomsRef);
    const roomId = newRoomRef.key.substring(0, 6).toUpperCase();
    
    this.currentUserId = Date.now().toString();
    this.playerName = playerName;
    this.playerTeam = team;
    this.currentRoomId = roomId;

    const roomData = {
      id: roomId,
      host: this.currentUserId,
      status: 'waiting',
      createdAt: Date.now(),
      settings: settings || {
        maxPlayers: 4,
        timePerQuestion: 15,
        totalQuestions: 20
      },
      players: {
        [this.currentUserId]: {
          name: playerName,
          team: team,
          isReady: false,
          score: 0,
          joinedAt: Date.now()
        }
      },
      gameState: {
        currentQuestionIndex: 0,
        currentTurn: null,
        bellPressed: false,
        answeringTeam: null,
        scores: { A: 0, B: 0, C: 0, D: 0 }
      }
    };

    await set(ref(database, `rooms/${roomId}`), roomData);
    this.listenToRoom(roomId);
    
    return roomId;
  }

  // THAM GIA PHÒNG
  async joinRoom(roomId, playerName, team) {
    const roomRef = ref(database, `rooms/${roomId}`);
    
    // Kiểm tra phòng tồn tại
    const snapshot = await get(roomRef);
    if (!snapshot.exists()) {
      throw new Error('Phòng không tồn tại');
    }

    const room = snapshot.val();
    const playerCount = Object.keys(room.players || {}).length;
    
    if (playerCount >= room.settings.maxPlayers) {
      throw new Error('Phòng đã đầy');
    }

    if (room.status !== 'waiting') {
      throw new Error('Game đã bắt đầu');
    }

    this.currentUserId = Date.now().toString();
    this.playerName = playerName;
    this.playerTeam = team;
    this.currentRoomId = roomId;

    await update(ref(database, `rooms/${roomId}/players/${this.currentUserId}`), {
      name: playerName,
      team: team,
      isReady: false,
      score: 0,
      joinedAt: Date.now()
    });

    this.listenToRoom(roomId);
  }

  // LẮNG NGHE THAY ĐỔI PHÒNG
  listenToRoom(roomId) {
    const roomRef = ref(database, `rooms/${roomId}`);
    
    onValue(roomRef, (snapshot) => {
      const room = snapshot.val();
      
      if (!room) {
        console.log('Room deleted');
        return;
      }

      // Cập nhật UI
      this.updatePlayerList(room.players);
      this.updateGameState(room.gameState);
      
      if (room.currentQuestion) {
        this.displayQuestion(room.currentQuestion);
      }

      if (room.status === 'playing') {
        this.onGameStart(room);
      } else if (room.status === 'finished') {
        this.onGameFinish(room);
      }
    });

    // Lắng nghe sự kiện nhấn chuông
    const bellRef = ref(database, `rooms/${roomId}/gameState/bellPressed`);
    onValue(bellRef, (snapshot) => {
      if (snapshot.val()) {
        this.onBellPressed(roomId);
      }
    });
  }

  // SẴN SÀNG
  async toggleReady() {
    const playerRef = ref(database, `rooms/${this.currentRoomId}/players/${this.currentUserId}/isReady`);
    const snapshot = await get(playerRef);
    await set(playerRef, !snapshot.val());
  }

  // BẮT ĐẦU GAME
  async startGame() {
    const roomRef = ref(database, `rooms/${this.currentRoomId}`);
    const snapshot = await get(roomRef);
    const room = snapshot.val();

    // Kiểm tra host
    if (room.host !== this.currentUserId) {
      throw new Error('Chỉ host mới có thể bắt đầu');
    }

    // Load câu hỏi
    const questions = await this.loadQuestions(room.settings.totalQuestions);
    
    await update(roomRef, {
      status: 'playing',
      'gameState/currentQuestionIndex': 0,
      'gameState/currentTurn': Object.values(room.players)[0].team,
      currentQuestion: questions[0]
    });
  }

  // NHẤN CHUÔNG
  async ringBell() {
    const timestamp = Date.now();
    
    await update(ref(database, `rooms/${this.currentRoomId}/gameState`), {
      bellPressed: true,
      answeringTeam: this.playerTeam,
      answeringPlayer: this.playerName,
      bellTimestamp: timestamp
    });
  }

  // TRẢ LỜI
  async submitAnswer(answer) {
    const roomRef = ref(database, `rooms/${this.currentRoomId}`);
    const snapshot = await get(roomRef);
    const room = snapshot.val();

    const isCorrect = answer.toLowerCase().trim() === 
                     room.currentQuestion.answer.toLowerCase().trim();

    const scoreChange = isCorrect ? 10 : -5;
    const newScore = (room.gameState.scores[this.playerTeam] || 0) + scoreChange;

    await update(ref(database, `rooms/${this.currentRoomId}/gameState/scores`), {
      [this.playerTeam]: newScore
    });

    // Hiển thị kết quả
    this.showAnswerResult(isCorrect, room.currentQuestion.answer);

    // Chuyển câu tiếp theo sau 3 giây
    setTimeout(() => {
      this.nextQuestion();
    }, 3000);
  }

  // CÂU HỎI TIẾP THEO
  async nextQuestion() {
    const roomRef = ref(database, `rooms/${this.currentRoomId}`);
    const snapshot = await get(roomRef);
    const room = snapshot.val();

    const nextIndex = room.gameState.currentQuestionIndex + 1;
    const questions = await this.loadQuestions(room.settings.totalQuestions);

    if (nextIndex >= questions.length) {
      // Kết thúc game
      await update(roomRef, { status: 'finished' });
    } else {
      await update(roomRef, {
        'gameState/currentQuestionIndex': nextIndex,
        'gameState/bellPressed': false,
        'gameState/answeringTeam': null,
        currentQuestion: questions[nextIndex]
      });
    }
  }

  // RỜI PHÒNG
  async leaveRoom() {
    if (!this.currentRoomId || !this.currentUserId) return;

    const playerRef = ref(database, `rooms/${this.currentRoomId}/players/${this.currentUserId}`);
    await remove(playerRef);

    // Nếu là host, chuyển host hoặc xóa phòng
    const roomRef = ref(database, `rooms/${this.currentRoomId}`);
    const snapshot = await get(roomRef);
    const room = snapshot.val();

    if (room && room.host === this.currentUserId) {
      const remainingPlayers = Object.keys(room.players || {}).filter(id => id !== this.currentUserId);
      
      if (remainingPlayers.length === 0) {
        await remove(roomRef);
      } else {
        await update(roomRef, { host: remainingPlayers[0] });
      }
    }
  }

  // HELPER FUNCTIONS
  async loadQuestions(count) {
    const questionsRef = ref(database, 'questions');
    const snapshot = await get(questionsRef);
    const allQuestions = snapshot.val();
    
    return Object.values(allQuestions).slice(0, count);
  }

  updatePlayerList(players) {
    // Cập nhật UI danh sách người chơi
  }

  updateGameState(gameState) {
    // Cập nhật UI điểm số, lượt chơi
  }

  displayQuestion(question) {
    // Hiển thị câu hỏi
  }

  onBellPressed(roomId) {
    // Phát âm thanh chuông
    const bellSound = document.getElementById('bell-sound');
    bellSound.play();
  }

  onGameStart(room) {
    // Chuyển sang màn hình chơi
  }

  onGameFinish(room) {
    // Hiển thị kết quả
  }

  showAnswerResult(isCorrect, correctAnswer) {
    // Hiển thị kết quả trả lời
  }
}

export default GameManager;

Tiêu chí	WebSocket (Socket.io)	Firebase Realtime DB
Độ trễ	10-50ms	50-200ms
Chi phí	Chi phí server riêng	Free đến 100 concurrent, sau đó $5/GB
Khả năng mở rộng	Cần cấu hình Load Balancer	Tự động scale
Độ phức tạp	Cao (cần setup server)	Thấp (chỉ cần config)
Offline support	Không	Có (tự động sync)
Bảo mật	Tự implement	Firebase Rules
Phù hợp	Game realtime cao cấp	MVP, prototype nhanh
